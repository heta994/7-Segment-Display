4-Digit 7 Segment Display using TM1637

This Arduino project uses the TM1637 module to display numbers and temperature values on a 4-digit 7-segment display.

Arduino Code


#include <TM1637Display.h>
#define CLK 2
#define DIO 3
TM1637Display display = TM1637Display(CLK, DIO);
const uint8_t celsius[] = { SEG_A | SEG_B | SEG_F | SEG_G, SEG_A | SEG_D | SEG_E | SEG_F };

void setup() {
  display.clear();
  delay(1000);
}
void loop() {
  display.setBrightness(4);
  for (int i = 0; i < 101; i++) {
    display.showNumberDec(i);
    delay(50);
  }
  delay(1000);
  display.clear();
  int temperature = 24;
  display.showNumberDec(temperature, false, 2, 0);
  display.setSegments(celsius, 2, 2);
  delay(1000);
  while(1);
}
